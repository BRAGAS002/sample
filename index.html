JavaScript Cheat Sheet

1. Data Types

// Primitive Types
let x = 5;        // Number
let name = "John"; // String
let isActive = true; // Boolean
let value = null;  // Null
let data;          // Undefined
let sym = Symbol(); // Symbol
let big = 1234567890123456789012345678901234567890n; // BigInt

// Reference Types
let person = { name: "John", age: 30 }; // Object
let colors = ["red", "green", "blue"];  // Array
function greet() { console.log("Hello!"); } // Function

2. Operators

// Arithmetic
let sum = 5 + 3;
let diff = 10 - 2;

// Comparison
let isEqual = 5 === 5;
let isGreater = 5 > 3;

// Logical
let result = true && false;

// Assignment
let x = 10;
x += 5; // x = x + 5;

// Ternary
let message = condition ? "Yes" : "No";

3. Control Flow

// if/else
if (condition) {
  // code
} else {
  // code
}

// switch
switch (expression) {
  case value1:
    // code
    break;
  default:
    // code
}

// for loop
for (let i = 0; i < 10; i++) {
  console.log(i);
}

// while loop
while (condition) {
  // code
}

4. Functions

// Declaration
function greet(name) {
  console.log("Hello, " + name);
}

// Expression
const greet = function(name) {
  console.log("Hello, " + name);
};

// Arrow Function
const greet = (name) => console.log("Hello, " + name);

5. Arrays

// Accessing Elements
let first = arr[0];

// Methods
arr.push("new item");  // Add to end
arr.pop();              // Remove from end
arr.shift();            // Remove from beginning
arr.unshift("new item"); // Add to beginning

arr.map(item => item * 2); // Transform elements
arr.filter(item => item > 5); // Filter elements
arr.forEach(item => console.log(item)); // Loop through array
arr.reduce((acc, item) => acc + item, 0); // Reduce array to value

6. Objects

// Accessing properties
let name = obj.name;
let name = obj["name"];

// Destructuring
const { name, age } = person;

7. DOM Manipulation

// Select Element
let element = document.getElementById("id");
let element = document.querySelector(".class");

// Modify Content
element.innerHTML = "text";
element.textContent = "text";

// Add Event Listener
element.addEventListener('click', function() {
  // code
});

8. Promises

// Creating a Promise
const myPromise = new Promise((resolve, reject) => {
  if (success) {
    resolve("Success!");
  } else {
    reject("Failure!");
  }
});

// Handling Promises
myPromise
  .then((result) => console.log(result))
  .catch((error) => console.log(error));

9. ES6+ Features

// Template Literals
let greeting = `Hello, ${name}!`;

// Destructuring Arrays
const [a, b] = [1, 2];

// Default Parameters
function greet(name = "Guest") {
  console.log("Hello, " + name);
}

10. Error Handling

// Try...Catch
try {
  // code that might throw an error
} catch (error) {
  console.log(error);
}

11. Asynchronous JavaScript

// setTimeout
setTimeout(() => {
  console.log("Hello after 1 second");
}, 1000);

// setInterval
setInterval(() => {
  console.log("Hello every 2 seconds");
}, 2000);

12. Classes

// Class Definition
class Person {
  constructor(name, age) {
    this.name = name;
    this.age = age;
  }
  greet() {
    console.log("Hello, " + this.name);
  }
}
